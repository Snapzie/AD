\section{Task 1}
Vi Bliver bedt om at give pseudokode for algoritmen GET-KTH-Key(x,k)
\begin{lstlisting}
	GET-KTH-KEY(x,k):
		if(k < 1 || k > x.size)
			return nil
		elif(x.size == 1)
			return x.key
		elif(x.size - x.right.size == k)
			return x.key
		elif(x.left.size >= k)
			return GET-KTH-KEY(x.left, k)
		else
			return GET-KTH-KEY(x.right, k - x.left.size)
\end{lstlisting}
