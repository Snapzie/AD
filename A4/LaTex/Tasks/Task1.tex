\section{Task 1}
Vi Bliver bedt om at give pseudokode for algoritmen GET-KTH-KEY(x,k)
\begin{lstlisting}
	GET-KTH-KEY(x,k):
		if(k < 1 || k > x.size + 1)
			return nil
		elif(x.size == 0)
			return x.key
		elif(x.size - x.right.size == k)
			return x.key
		elif(x.left.size + 1 >= k)
			return GET-KTH-KEY(x.left, k)
		else
			return GET-KTH-KEY(x.right, k - (x.size - x.right.size))
\end{lstlisting}
